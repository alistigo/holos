# Task: Scaffold @alistigo/artifact-plugin-api

## Description

Create the new leaf package `packages/alistigo-artifact-plugin-api` (npm
`@alistigo/artifact-plugin-api`) that defines the unified `AlistigoPlugin` interface,
the typed event bus, the plugin runtime, and the CDN dynamic-import loader. This is the
foundational package every other task in this epic depends on.

Use `bash scripts/new-package.sh alistigo-artifact-plugin-api` to scaffold, then replace
generated `src/index.ts` contents per the technical details below.

## Acceptance Criteria

- [ ] Package scaffolded at `packages/alistigo-artifact-plugin-api` with `project.json` (Nx targets only, no duplicate `package.json` scripts per repo convention)
- [ ] `src/types.ts` exports `AlistigoPluginEventMap`, `PluginEventName`, `PluginEventBus`, `PluginContext`, `AlistigoPlugin`
- [ ] `src/runtime.ts` exports `createPluginBus()` and `createPluginRuntime(plugins, baseContext, configByPlugin)`
- [ ] `src/loader.ts` exports `resolvePluginUrl(packageName)` and `loadPlugin(packageName)`
- [ ] `src/index.ts` barrel re-exports all of the above
- [ ] Every lifecycle hook invocation inside `createPluginRuntime` is individually wrapped in try/catch, logging via `ctx.logger.error` and continuing to the next plugin/hook on failure
- [ ] Plugins execute in `Object.entries(configByPlugin)` insertion order (documented in a code comment as a contract, not just an implementation detail)
- [ ] Unit tests: event bus on/emit/unsubscribe; runtime continues after a plugin's `setup()` throws; `resolvePluginUrl` produces the expected jsDelivr `@0`-pinned URL shape
- [ ] `pnpm build` and `pnpm build:typecheck` pass for this package
- [ ] `pnpm nx qa:lint` passes for this package

## Technical Details

```ts
// src/types.ts
export interface AlistigoPluginEventMap {
  "widget:displayed": { locale: string; storageType: string; version: string };
  "error:uncaught": { error: unknown; componentStack?: string };
}
export type PluginEventName = keyof AlistigoPluginEventMap;
export interface PluginEventBus {
  on<E extends PluginEventName>(event: E, handler: (payload: AlistigoPluginEventMap[E]) => void): () => void;
  emit<E extends PluginEventName>(event: E, payload: AlistigoPluginEventMap[E]): void;
}
export interface PluginContext {
  config: Record<string, unknown>;
  host: { packageName: string; version: string; locale: string; environment: string };
  logger: { info: (o: unknown, msg?: string) => void; error: (o: unknown, msg?: string) => void };
  on: PluginEventBus["on"];
  emit: PluginEventBus["emit"];
}
export interface AlistigoPlugin {
  name: string;
  setup?(ctx: PluginContext): void | Promise<void>;
  beforeMount?(ctx: PluginContext): void | Promise<void>;
  mounted?(ctx: PluginContext): void | Promise<void>;
  destroy?(ctx: PluginContext): void | Promise<void>;
  wrapRoot?(tree: import("react").ReactNode, ctx: PluginContext): import("react").ReactNode;
  dataShape?: unknown;
  render?: unknown;
  commands?: Record<string, (...args: unknown[]) => unknown>;
  events?: Record<string, unknown>;
}
```

```ts
// src/loader.ts
export function resolvePluginUrl(packageName: string): string {
  return `https://cdn.jsdelivr.net/npm/${packageName}@0/dist/index.js`;
}
export async function loadPlugin(packageName: string): Promise<AlistigoPlugin> {
  const mod = await import(/* @vite-ignore */ resolvePluginUrl(packageName));
  return mod.default as AlistigoPlugin;
}
```
